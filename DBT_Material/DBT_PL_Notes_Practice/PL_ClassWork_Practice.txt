CASES IN SQL-PL



DELIMITER //

CREATE PROCEDURE PLCASE()

BEGIN
	DECLARE X INT;
	SELECT BASIC INTO X FROM SEMP WHERE EMPNAME = 'GEORGE';
	
	CASE
	WHEN X > 5000 THEN
		INSERT INTO TEMPP VALUES (X, 'HIGH SALARY');
	ELSE
		INSERT INTO TEMPP VALUES (X, 'LOW SALARY');
	END CASE;
	
	

END //

CURSOR CLASS QUESTION:


CREATE TABLE CURSOREMP:

CREATE TABLE CURSOREMP(
    -> EMPNO INT,
    -> ENAME VARCHAR(15),
    -> SAL INT,
    -> DEPTNO INT
    -> );
    
    
DELIMITER //

CREATE PROCEDURE CURSOREMP()

BEGIN
	
	DECLARE A INT;
	DECLARE B VARCHAR(15);
	DECLARE C INT;
	DECLARE D INT;
	DECLARE X INT DEFAULT 0;
	DECLARE C1 CURSOR FOR SELECT * FROM CURSOREMP;
	OPEN C1;
	WHILE X < 5 DO
		FETCH C1 INTO A,B,C,D;
		INSERT INTO TEMPP VALUES(A,B);
		SET X = X+1;
	END WHILE;
	CLOSE C1;

END //

DELIMITER ;
